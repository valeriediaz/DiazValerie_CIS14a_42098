		<script>
			//	create & initialize variables
			var	str, name, color, month,	//	string variables
				correct, ans, ansYr,		//	boolean variables
				num=0, year=0, i=0;			//	integer variables
			
		//	prompt user for information
			//	check if the user wants to proceed using a do/while loop
			do{
				//	get string
				name = prompt("What is your name?");
				//	get boolean
				correct = confirm("So, I will call you " + name + ", then.\n\n" + 
					"Click 'Cancel' to re-enter your name.");
			}while(!correct);
			//	get integer
			num = prompt("What is your lucky number?");
			//	get string
			color = prompt("What is your favorite color?");
			//	get string
			month = prompt("What month were you born?");
		//	check if user wants to answer using an if/else statement
			//	get boolean
			ansYr = confirm("Can I ask you what year you were born?\n\n" + 
				"Click 'Cancel' if you wish not to answer.");
			if(ansYr == true){
				//	get integer
				year = prompt("What year were you born?");
			}else{
				alert("I respect your privacy.");
			};
		//	check what info the user wants to display using an if/else statement
			//	get boolean
			ans = confirm("Can I display your information on the screen?\n\n" + 
				"Click 'Cancel' if you wish to keep your information private.");
			//	write an html table into a string variable
			if(ans == true){
				str = "<table>";
					str += "<tr>";	//display name
						str += "<td align='right'><b>Name: </b></td>";
						str += ("<td>" + name + "</td>");
					str += "</tr>";
					str += "<tr>";	//display number
						str += "<td align='right'><b>Number: </b></td>";
						str += ("<td>" + num + "</td>");
					str += "</tr>";
					str += "<tr>";	//display color
						str += "<td align='right'><b>Color: </b></td>";
						str += ("<td>" + color + "</td>");
					str += "</tr>";
					str += "<tr>";	//display month
						str += "<td align='right'><b>Month: </b></td>";
						str += ("<td>" + month + "</td>");
					str += "</tr>";
					str += "<tr>";	//display year
						str += "<td align='right'><b>Year: </b></td>";
						//	determine whether to display the year using an if/else statement
						if(ansYr == true){
							str += ("<td>" + year + "</td>");
						}else{
							str += "<td></td>";
						}
					str += "</tr>";
				str += "</table>";
			}else{
				str = "<h1 style='color:red;'>CONFIDENTIAL!</h1>";
			};
			//	display the string by writing to the webpage
			document.write(str);
		</script>
